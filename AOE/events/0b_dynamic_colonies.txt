country_event = {
	title = "New Colony"
	desc = "We happened to be in possession of new overseas land, it's time to establish formal administration."
	#picture = "Education"
	#major = yes

	id = 9999819

	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	immediate = {
	          name = "Ok"
			  any_country = {
			                 limit = {
			                          AND = { 
				               	             tag = FROM
                                             #exists = no							  
				               	      }
			                 }
	                         any_owned = {
				               		       limit = {
				               		                AND = {
				               		                       is_core = UTI
				               		                       is_overseas = yes
														   NOT = { is_core = FROM }
				               			    		}
				               		       }
				               		       secede_province = THIS						
			                 }
			  }			  
	}
	option = {
	          name = "Ok"
	          UTI = { country_event = 9999817 }
			  clr_global_flag = making_colonies
			  clr_global_flag = colony_already_established
	}
}
