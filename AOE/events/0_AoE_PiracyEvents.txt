	desc = EVTDESC69691
	picture = piracy_golden_ended
	fire_only_once = yes
	trigger = {
		year = 1730
	}
	mean_time_to_happen = {
		months = 0
	}
	
	option = {
		name = "Another sun dips below the waves"
	    any_owned = {
            limit = {
				has_province_modifier = piracy_harsh
				OR = {
					continent = west_africa
					continent = south_west_africa
					continent = north_america
					continent = south_america
				}
			}
			remove_province_modifier = piracy_harsh
			add_province_modifier = {
				name = piracy_moderate
				duration = -1
			}
		}
		set_global_flag = golden_age_piracy_over
	}
}

	id = 69692
	title = "Piracy Victorious!"
	desc = "We have liberated former naval bases, piratical hotspots or centres of trade that we must integrate into our free Pirate Republic!"
	picture = piracy_conquest
	trigger = {
		government = pirate_republic
		any_owned_province = {
			OR = {
				AND = {
					has_building = naval_base
					NOT = { has_province_modifier = piracy_spoils }
				}
				has_province_modifier = piracy_harsh
			}
		}
	}
	mean_time_to_happen = { months = 0 }
	
	option = {
		name = "Hail Liberty!"
	    any_owned = {
            limit = {
				OR = {
					AND = {
						has_building = naval_base
						NOT = { has_province_modifier = piracy_spoils }
					}
					AND = {
						has_province_modifier = potential_trade_post
						NOT = { has_province_modifier = piracy_spoils }
					}
					AND = {
						has_province_modifier = trade_centre
						NOT = { has_province_modifier = piracy_spoils }
					}
					has_province_modifier = piracy_harsh
				}
			}
			add_province_modifier = {
				name = piracy_spoils
				duration = -1
			}
		}
		any_owned = {
			remove_province_modifier = piracy_harsh
			remove_province_modifier = piracy_moderate
			remove_province_modifier = piracy_mild
		}
	}
